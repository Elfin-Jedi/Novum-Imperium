# -*- ck2.decisions -*-

decisions = {
	conscript_merchant_ships = {
		only_playable = yes
		is_high_prio = yes
		ai_check_interval = 12

		potential = {
			year = 1000
			war = yes # EMF: Don't show this annoying thing while at peace
			NOT = { has_earmarked_regiments = conscripted_merchant_ships }
			has_overseas_holdings = yes
			emf_hiprio_decision_basic_potential = yes
		}
		allow = {
			war = yes
			trigger_if = {
				limit = { ai = no }
				custom_tooltip = {
					text = NEEDS_50_GOLD_COST
					wealth >= 50
				}
			}
			year >= 1000
		}
		effect = {
			if = {
				limit = { ai = no }
				wealth = -50
			}
			spawn_fleet = {
				province = closest # closest sea zone
				owner = ROOT
				disband_on_peace = yes
				earmark = conscripted_merchant_ships
				troops = {
					galleys = { 20 20 }
				}
			}
		}
		ai_will_do = {
			factor = 1
		}
	}

	raise_tribal_units = {
		only_playable = yes
		is_high_prio = yes
		is_mercenary = yes
		ai_check_interval = 12

		potential = {
			NOT = { has_dlc = "Legacy of Rome" }
			is_tribal = yes
			is_landed = yes
			OR = {
				any_war = {
					OR = {
						any_defender = {
							character = ROOT
							is_primary_war_defender = yes
						}
						any_attacker = {
							character = ROOT
							is_primary_war_attacker = yes
						}
					}
				}
			}
		}
		allow = {
			war = yes
			custom_tooltip = { text = NEEDS_500_PRESTIGE_COST hidden_tooltip = { prestige = 500 } }
			custom_tooltip = {
				text = emf_ctt_has_not_raised_tribal_army_for_3yr
				hidden_tooltip = {
					NOT = { has_character_modifier = raise_tribal_units_timer }
				}
			}
		}
		effect = {
			prestige = -500
			capital_scope = {
				PREV = {
					spawn_unit = {
						owner = ROOT
						province = PREV
						home = PREV
						match_character = ROOT
						match_mult = 1.0
						match_min = 500
						match_max = 2500
						attrition = 1.0
						disband_on_peace = yes
						can_toggle_looting = no
					}
				}
			}
			add_character_modifier = {
				name = raise_tribal_units_timer
				duration = 1095
			}
		}
		ai_will_do = {
			factor = 0 # decisions with the "is_mercenary" will not be calculated from the script
		}
	}

	raise_defensive_unit = {
		only_playable = yes
		is_high_prio = yes
		is_mercenary = yes
		ai_check_interval = 12

		potential = {
			has_dlc = "The Old Gods"
			NOT = { has_dlc = "Legacy of Rome" }
			is_tribal = yes
			is_landed = yes
			primary_title = {
				OR = {
					has_law = clanic_temple_control_2
					has_law = clanic_temple_control_3
					has_law = clanic_temple_control_4
				}
			}
		}

		allow = {
			is_primary_war_defender = yes
			custom_tooltip = { text = NEEDS_300_PIETY_COST hidden_tooltip = { piety = 300 } }
			custom_tooltip = {
				text = emf_ctt_has_not_raised_tribal_army_for_3yr
				hidden_tooltip = {
					NOT = { has_character_modifier = raise_tribal_units_timer }
				}
			}
			custom_tooltip = {
				text = raise_defensive_unit_tooltip

				hidden_tooltip = {
					any_war = {
						defender = { character = ROOT }

						#Defending against heathen
						OR = {
							#Steppe religion
							AND = {
								ROOT = {
									steppe_religion_trigger = yes
								}
								attacker = {
									steppe_religion_trigger = no
								}
							}

							#Messianic soul against any other religion
							AND = {
								ROOT = {
									messianic_soul_trigger = yes
								}
								attacker = {
									NOT = {
										religion = ROOT
									}
								}
							}
						}
					}
				}
			}
		}

		effect = {
			piety = -300
			capital_scope = {
				ROOT = {
					spawn_unit = {
						owner = ROOT
						home = PREV
						province = PREV
						match_character = ROOT
						match_mult = 1.0
						match_min = 500
						match_max = 2500
						disband_on_peace = yes
						attrition = 1.0
						can_toggle_looting = no
						earmark = defensive_pagan_troops
					}
				}
			}
			add_character_modifier = {
				name = raise_tribal_units_timer
				duration = 1095
			}
		}
		ai_will_do = {
			factor = 0
		}
	}
}
