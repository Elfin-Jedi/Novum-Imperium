# Messianic Revolt
messianic_revolt = {
	name = CB_NAME_MESSIANIC_REVOLT
	war_name = WAR_NAME_MESSIANIC_REVOLT
	sprite = 7
	truce_days = 365
	hostile_against_others = yes
	attacker_can_call_allies = no
	defender_can_call_allies = no
	can_ask_to_join_war = no
	#check_de_jure_tier = DUKE # comment out if they don't work
	
	can_use = {
		ROOT = {
			has_character_flag = messianic_rebel
			custom_tooltip = {
				text = NOT_BUSY
				hidden_tooltip = {
					is_inaccessible_trigger = no
				}
			}
		}
	}
	
	on_success = {
		FROM = { prestige = 500} 
		ROOT = { 
			clr_character_flag = messianic_rebel
			character_event = { id = luxrevolt.1064 }	# The realm becomes a theocracy to prevent further corruption of the true path.
			#imprison = FROM
		}

	}

	on_success_title = {
		custom_tooltip = {
			text = tribal_invasion_succ_tip
			hidden_tooltip = {
				ROOT = {
					vassalize_or_take_under_title = {
						title = PREV
						enemy = FROM
						same_religion = yes # Only vassalize rulers of my religion
						is_religious = yes
						type = revolt
					}
					if = {
						limit = {
							num_of_count_titles = 5
							PREV = { 
								OR = {
									has_holder = no 
									holder_scope = {
										character = FROM
									}
								}
							}
						}
						usurp_title = { target = PREV type = revolt }
					}
				}
			}
		}
	}

	on_fail_title = {
		FROM = {
			prestige = 100
			hidden_tooltip = {
				opinion = {
					modifier = opinion_rebel_traitor 
					who = ROOT
					months = 1200
				}
			}
		}
		ROOT = {
			clear_wealth = yes
			imprison = FROM
			clr_character_flag = messianic_rebel
			hidden_tooltip = {
				add_character_modifier = { 
					name = broken_spirit
					days = -1
				}
			}
		}
	}

	on_reverse_demand = {
		ROOT = {
			clear_wealth = yes
			imprison = FROM 
			clr_character_flag = messianic_rebel
			hidden_tooltip = {
				add_character_modifier = { 
					name = broken_spirit
					days = -1
				}
			}
		}
		FROM = {
			prestige = 100
			piety = 200
			hidden_tooltip = {
				opinion = {
					modifier = opinion_rebel_traitor 
					who = ROOT
					months = 1200
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = 100
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 150
	}
}
