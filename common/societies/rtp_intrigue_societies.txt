#Norse
rtp_intrigue_norse = {
	sound = assassins_interface
	society_ranks_gfx = devil_ranks
	primary_attribute = intrigue
	indestructible = yes
	society_influence = no
	opinion_to_other_members = 5
	opinion_per_rank_above = 5
	member_score_per_attribute = 1
	member_score_per_rank = 2
	
	active = {
		always = yes
	}

	can_join_society = {
		intrigue = 5
		lower_tier_than = KING
		rtp_is_capable_adult = yes
		OR = {
			religion_group = norsegerman_group
			religion_group = pagan_group
		}
		NOR = {
			AND = {
				ai = yes
				is_society_rank_full = {
					society = rtp_intrigue_norse
					rank = 1
				}
			}
			custom_tooltip = {
				text = "rtp_got_kicked_ct"
				has_character_flag = rtp_kicked_from_intrigue_society #Bye Felicia!
			}
		}
	}

	show_society = {
		OR = {
			religion_group = norsegerman_group
			religion_group = pagan_group
		}
	}

	potential = {
		lower_tier_than = KING
		OR = {
			religion_group = norsegerman_group
			religion_group = pagan_group
		}
	}

	society_rank = {
		level = 1
		limit = 50
		startup_limit = 30
		modifier = {
			murder_plot_power_modifier = 0.05
		}
		decisions = {
			rtp_intrigue_p1_td
		}
	}
	
	society_rank = {
		level = 2
		limit = 30
		startup_limit = 30
		modifier = {
			murder_plot_power_modifier = 0.1
			intrigue = 1
			monthly_character_prestige = -0.1
		}
		decisions = {
			rtp_intrigue_p2_td
			rtp_intrigue_pp2_td
		}
	}
	
	society_rank = {
		level = 3
		limit = 15
		startup_limit = 15
		modifier = {
			murder_plot_power_modifier = 0.15
			intrigue = 2
			monthly_character_prestige = -0.1
			general_opinion = -5
		}
		decisions = {
			rtp_intrigue_p3_sd
		}
	}

	society_rank = {
		level = 4
		limit = 5
		startup_limit = 5
		modifier = {
			murder_plot_power_modifier = 0.2
			intrigue = 3
			monthly_character_prestige = -0.1
			general_opinion = -10
		}
		decisions = {
			rtp_intrigue_p4_sd
			rtp_intrigue_mercenary_sd
			rtp_intrigue_privateer_sd
		}
	}

	monthly_currency_gain = {
		name = currency_name_rtp_intrigue

		per_attribute = {
			name = intrigue
			value = 1
		}
	}

	member_score = {
		value = 1

		modifier = {
			factor = 2

			custom_tooltip = {
				text = "rtp_is_a_noble_ct"

				OR = {
					AND = {
						ai = yes
						is_ruler = no

							dynasty = none
						}
					}
					AND = {
						ai = no
						rtp_is_landless = yes
						NOT = {
							dynasty = 19930
						}
					}
				}
			}
		}

		modifier = {
			factor = 3

			custom_tooltip = {
				text = "rtp_is_a_baron_ct"

				OR = {
					tier = BARON
					rtp_is_baron = yes
				}
			}

		modifier = {
			factor = 4

			custom_tooltip = {
				text = "rtp_is_a_count_ct"

				tier = COUNT
				is_landed = yes
			}
		}

		modifier = {
			factor = 5

			custom_tooltip = {
				text = "rtp_is_a_duke_ct"

				tier = DUKE
				is_landed = yes
			}
		}
	}
	
	startup_populate = {
		trigger = {
			ai = yes
			intrigue = 5
			is_female = no
			lower_tier_than = KING
			rtp_is_capable_adult = yes
			is_in_society = no
			OR = {
				religion_group = norsegerman_group
