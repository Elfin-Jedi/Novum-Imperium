character_event = { #Who are you taking with you?
	id  = DAQIN.00205
	desc = DAQIN.00205.D
	picture = GFX_evt_recieve_letter
	
	only_playable = yes
	
	is_triggered_only = yes
	
	option = { #Immediate family
		name = DAQIN.00205.OPTA
		trigger = { any_close_relative = { is_alive = yes } }#Need a spouse, sibling, grand/parent or grand/child to take with you
		any_close_relative = {
			opinion = {
				who = ROOT
				modifier = opinion_family_love
			}
		}
		scaled_wealth = -0.2
	}
	option = { #Extended family (all dynasty members in your realm)
		name = DAQIN.00205.OPTB
		trigger = { num_of_dynasty_members = 10 } #Must have a large dynasty
		any_dynasty_member = {
			limit = {
				NOT = { character = ROOT }
			}
			opinion = {
				who = ROOT
				modifier = opinion_generous_relative
			}
		}
		scaled_wealth = -0.4
	}
	option = { #Friends and 'friends'
		name = DAQIN.00205.OPTC
		trigger = { 
			OR = {
				num_of_friends = 1
				AND = {
					NOT = { any_close_relative = { is_alive = no } } #No family
					num_of_lovers = 1 #But have a lover - prevents rare softlock for family-less unmarried chars
				}
			}
		}
		set_variable = { which = AmountOfFriends value = 0 } #Subtracts 20 wealth per friend. ##TODO: Implement this for all these decisions?
		export_to_variable = {
			which = AmountOfFriends
			value = num_of_friends
		}
		while = { #Costs 20 gold per friend
			limit = {
				check_variable = {
					which = AmountOfFriends
					value >= 1
				}
			}
			subtract_variable = {
				which = AmountOfFriends
				value = 1
			}
			wealth = -20
		}
		scaled_wealth = -0.1
		random = {
			chance = 25
			add_trait = gregarious
		}
	}
	option = { #I have nobody...
		name = DAQIN.00205.OPTD
		trigger = {
			num_of_dynasty_members == 1
			is_married = no
			num_of_children < 1
			num_of_friends < 1
			num_of_lovers < 1
		}
		clr_character_flag = western_holiday
		clr_character_flag = eastern_holiday
		character_event = { id = DAQIN.00206 days = 10 }
		break = yes
	}
	after = { #TODO: Clean this up, it's so horribly inefficient
		character_event = { id = DAQIN.00209 days = 30 } #Holiday set off event
		if = { limit = { spouse = { trait = homosexual } } character_event = { id = DAQIN.00207 days = 10 } } #Gay spouse event
		if = { limit = { num_of_lovers < 1 } break = yes }
		else_if = { limit = { is_married = yes num_of_lovers == 1 } character_event = { id = DAQIN.00208 days = 15 } } #Lover conflict
		else_if = { limit = { is_married = no  num_of_lovers == 1 } character_event = { id = DAQIN.002051 days = 240 } } #Marry lover event
	}
}
