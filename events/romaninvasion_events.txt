# Claim on HRE
narrative_event = {
	id = RIA.73
	title = EVTTITLE_RI_1
	desc = EVTDESC_RI_1
	picture = GFX_evt_roman_conquest
	
	only_playable = yes
	religion_group = pagan_group
	only_independent = yes
	only_capable = yes
	min_age = 16
	prisoner = no 
	war = no

	trigger = {
		year = 1070
		OR = {
			AND = {
				OR = {
					culture = roman
					culture = greek
					culture = italian
				}
				NOT = {
					religion_group = christian
				}
				NOT = { year = 1090 }
				is_feudal = yes
			}
		}
		has_landed_title = e_roman_empire
		has_global_flag = roman_arrival
		has_global_flag = roman_wars3
		e_hre = {
			holder_scope = {
				religion_group = christian
			}
		}
	}
	
	mean_time_to_happen = {
		months = 5
	}
	
	option = {
		name = EVTOPTA_CM_1100 # Attack
		ai_chance = { factor = 100 }
		piety = 50
		add_claim = e_hre
		e_hre = {
			holder_scope = {
				reverse_war = {
					target = ROOT
					casus_belli = claim
					thirdparty_title = PREV
					infamy = 0
				}
			}
		}
		set_global_flag = roman_wars
	}
	option = {
		name = EVTOPTB_CM_1100 # Not now
		ai_chance = { factor = 0 }
		prestige = -100
		piety = -100
		set_global_flag = roman_wars
	}
}
